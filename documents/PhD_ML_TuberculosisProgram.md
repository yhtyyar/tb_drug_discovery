#### 6.4 –ü–†–ò–ú–ï–† –ö–û–î–ê: –ü–æ–ª–Ω—ã–π pipeline

**File: tb_discovery_pipeline.py**

```python
"""
–ü–æ–ª–Ω—ã–π pipeline –¥–ª—è –¢–ë drug discovery —Å ML
–£—á–µ–±–Ω—ã–π –ø—Ä–∏–º–µ—Ä: QSAR + GNN + Docking
"""

import pandas as pd
import numpy as np
from rdkit import Chem
from rdkit.Chem import AllChem, Descriptors
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split
from sklearn.metrics import r2_score, mean_squared_error
import torch
import torch.nn as nn
from torch_geometric.data import Data, DataLoader
from torch_geometric.nn import GCNConv, global_mean_pool
import warnings
warnings.filterwarnings('ignore')

# ====================== PART 1: DATA LOADING ======================

def load_chembl_tb_data():
    """
    –ó–∞–≥—Ä—É–∂–∞–µ—Ç –¢–ë –∏–Ω–≥–∏–±–∏—Ç–æ—Ä—ã –∏–∑ ChEMBL (—Å–∏–º—É–ª—è—Ü–∏—è)
    –í —Ä–µ–∞–ª—å–Ω–æ—Å—Ç–∏ –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è ChEMBL API
    """
    # –°–∏–º—É–ª–∏—Ä—É–µ–º –¥–∞—Ç–∞—Å–µ—Ç: [SMILES, IC50_nM, source]
    data = {
        'SMILES': [
            'CC(=O)Nc1cccc(C(F)(F)F)c1',
            'Cc1cc(C)cc(Nc2ccccc2C(=O)O)c1',
            'CN1CC(N)C(=O)C(C(=O)Nc2ccccc2)=C1',
            # ... –≤ —Ä–µ–∞–ª—å–Ω–æ—Å—Ç–∏ —Å–æ—Ç–Ω–∏ —Å–æ–µ–¥–∏–Ω–µ–Ω–∏–π
        ],
        'IC50_nM': [10, 50, 25],  # –ò–Ω–≥–∏–±–∏—Ç–æ—Ä–Ω–∞—è –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å
        'target': ['InhA', 'InhA', 'rpoB']
    }
    return pd.DataFrame(data)

# ====================== PART 2: FEATURE ENGINEERING ======================

def calculate_rdkit_descriptors(smiles):
    """–†–∞—Å—á–µ—Ç –º–æ–ª–µ–∫—É–ª—è—Ä–Ω—ã—Ö –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä–æ–≤ –∏–∑ SMILES"""
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None
    
    descriptors = {
        'MW': Descriptors.MolWt(mol),
        'LogP': Descriptors.MolLogP(mol),
        'TPSA': Descriptors.TPSA(mol),
        'HBA': Descriptors.NumHAcceptors(mol),
        'HBD': Descriptors.NumHDonors(mol),
        'RotBonds': Descriptors.NumRotatableBonds(mol),
        'RingCount': Descriptors.RingCount(mol),
        'AromaticRings': Descriptors.NumAromaticRings(mol)
    }
    return descriptors

def mol_to_graph(smiles):
    """–ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ SMILES –≤ –≥—Ä–∞—Ñ –¥–ª—è GNN"""
    mol = Chem.MolFromSmiles(smiles)
    if mol is None:
        return None
    
    # –£–∑–ª—ã = –∞—Ç–æ–º—ã, —Ä–µ–±—Ä–∞ = —Å–≤—è–∑–∏
    num_atoms = mol.GetNumAtoms()
    
    # –ú–∞—Ç—Ä–∏—Ü–∞ —Å–º–µ–∂–Ω–æ—Å—Ç–∏
    adjacency = Chem.GetAdjacencyMatrix(mol)
    edge_indices = torch.nonzero(torch.tensor(adjacency)).t().contiguous()
    
    # –ü—Ä–∏–∑–Ω–∞–∫–∏ —É–∑–ª–æ–≤ (–∞—Ç–æ–º—ã)
    node_features = []
    for atom in mol.GetAtoms():
        features = [
            atom.GetAtomicNum(),  # –ê—Ç–æ–º–Ω—ã–π –Ω–æ–º–µ—Ä
            atom.GetDegree(),     # –°—Ç–µ–ø–µ–Ω—å —Å–≤—è–∑–∞–Ω–Ω–æ—Å—Ç–∏
            atom.GetFormalCharge(),  # –ó–∞—Ä—è–¥
            int(atom.GetIsAromatic())  # –ê—Ä–æ–º–∞—Ç–∏—á–µ–Ω –ª–∏
        ]
        node_features.append(features)
    
    x = torch.tensor(node_features, dtype=torch.float)
    
    return Data(x=x, edge_index=edge_indices)

# ====================== PART 3: MODEL TRAINING ======================

class MoleculeGNN(nn.Module):
    """Graph Neural Network –¥–ª—è –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏"""
    
    def __init__(self, input_dim=4, hidden_dim=32, output_dim=1):
        super(MoleculeGNN, self).__init__()
        self.conv1 = GCNConv(input_dim, hidden_dim)
        self.conv2 = GCNConv(hidden_dim, hidden_dim)
        self.linear = nn.Linear(hidden_dim, output_dim)
    
    def forward(self, x, edge_index, batch):
        x = self.conv1(x, edge_index)
        x = torch.relu(x)
        x = self.conv2(x, edge_index)
        x = global_mean_pool(x, batch)  # –ì–ª–æ–±–∞–ª—å–Ω–∞—è –∞–≥—Ä–µ–≥–∞—Ü–∏—è
        x = self.linear(x)
        return x

def train_qsar_baseline(df):
    """–û–±—É—á–µ–Ω–∏–µ –∫–ª–∞—Å—Å–∏—á–µ—Å–∫–æ–π QSAR (Random Forest)"""
    
    # –†–∞—Å—á–µ—Ç –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä–æ–≤
    features_list = []
    for smiles in df['SMILES']:
        feat = calculate_rdkit_descriptors(smiles)
        if feat is not None:
            features_list.append(feat)
    
    X = pd.DataFrame(features_list)
    y = np.log10(df['IC50_nM'].values)  # Log scale (–∞–∫—Ç–∏–≤–Ω–æ—Å—Ç—å)
    
    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)
    
    # –û–±—É—á–µ–Ω–∏–µ
    model = RandomForestRegressor(n_estimators=100, max_depth=10)
    model.fit(X_train, y_train)
    
    # –û—Ü–µ–Ω–∫–∞
    y_pred = model.predict(X_test)
    r2 = r2_score(y_test, y_pred)
    rmse = np.sqrt(mean_squared_error(y_test, y_pred))
    
    print(f"QSAR (Random Forest) R¬≤ = {r2:.3f}, RMSE = {rmse:.3f}")
    return model

def train_gnn_model(df, epochs=50, learning_rate=0.01):
    """–û–±—É—á–µ–Ω–∏–µ GNN –º–æ–¥–µ–ª–∏"""
    
    # –ü—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –≤ –≥—Ä–∞—Ñ—ã
    graphs = []
    y_list = []
    for idx, row in df.iterrows():
        graph = mol_to_graph(row['SMILES'])
        if graph is not None:
            graphs.append(graph)
            y_list.append(np.log10(row['IC50_nM']))
    
    # –°–æ–∑–¥–∞–Ω–∏–µ –¥–∞—Ç–∞—Å–µ—Ç–∞
    for graph, y in zip(graphs, y_list):
        graph.y = torch.tensor([y], dtype=torch.float)
    
    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ
    split_idx = int(0.8 * len(graphs))
    train_graphs = graphs[:split_idx]
    test_graphs = graphs[split_idx:]
    
    # DataLoader
    train_loader = DataLoader(train_graphs, batch_size=8, shuffle=True)
    test_loader = DataLoader(test_graphs, batch_size=8)
    
    # –ú–æ–¥–µ–ª—å
    model = MoleculeGNN(input_dim=4, hidden_dim=32, output_dim=1)
    optimizer = torch.optim.Adam(model.parameters(), lr=learning_rate)
    criterion = nn.MSELoss()
    
    # –û–±—É—á–µ–Ω–∏–µ
    for epoch in range(epochs):
        model.train()
        total_loss = 0
        for batch in train_loader:
            optimizer.zero_grad()
            out = model(batch.x, batch.edge_index, batch.batch)
            loss = criterion(out, batch.y.unsqueeze(1))
            loss.backward()
            optimizer.step()
            total_loss += loss.item()
        
        if (epoch + 1) % 10 == 0:
            print(f"Epoch {epoch+1}/{epochs}, Loss: {total_loss:.4f}")
    
    # –û—Ü–µ–Ω–∫–∞
    model.eval()
    y_pred_list = []
    y_true_list = []
    with torch.no_grad():
        for batch in test_loader:
            out = model(batch.x, batch.edge_index, batch.batch)
            y_pred_list.extend(out.squeeze().numpy())
            y_true_list.extend(batch.y.numpy())
    
    r2 = r2_score(y_true_list, y_pred_list)
    rmse = np.sqrt(mean_squared_error(y_true_list, y_pred_list))
    
    print(f"GNN R¬≤ = {r2:.3f}, RMSE = {rmse:.3f}")
    
    return model

# ====================== MAIN ======================

if __name__ == "__main__":
    # –ó–∞–≥—Ä—É–∑–∫–∞ –¥–∞–Ω–Ω—ã—Ö
    df = load_chembl_tb_data()
    print(f"–ó–∞–≥—Ä—É–∂–µ–Ω–æ {len(df)} —Å–æ–µ–¥–∏–Ω–µ–Ω–∏–π")
    
    # QSAR baseline
    print("\n=== QSAR (Random Forest) ===")
    qsar_model = train_qsar_baseline(df)
    
    # GNN training
    print("\n=== GNN Model ===")
    gnn_model = train_gnn_model(df, epochs=50)
    
    print("\n–ü–∞–π–ø–ª–∞–π–Ω –∑–∞–≤–µ—Ä—à–µ–Ω!")
```
